    <script>

        function calculatePrice() {

            // Değerleri al ve sayıya çevir

            const medineNights = parseInt(document.getElementById('medineNights').value) || 0;

            const mekkeNights = parseInt(document.getElementById('mekkeNights').value) || 0;

            const groupSize = parseInt(document.getElementById('groupSize').value) || 1;

            const priceDbl = parseFloat(document.getElementById('priceDbl').value) || 0;

            const priceTrpl = parseFloat(document.getElementById('priceTrpl').value) || 0;

            const priceQuad = parseFloat(document.getElementById('priceQuad').value) || 0;

            const totalBusCost = parseFloat(document.getElementById('busCost').value) || 0;

            const totalHighCost = parseFloat(document.getElementById('highCost').value) || 0;

            const totalGmcCost = parseFloat(document.getElementById('gmcCost').value) || 0;

            const perPersonTrainCost = parseFloat(document.getElementById('trainCost').value) || 0;

           

            // Kişi başı ulaşım maliyetlerini hesapla

            const perPersonBusCost = totalBusCost / groupSize;

            const perPersonHighCost = totalHighCost / groupSize;

            const perPersonGmcCost = totalGmcCost / groupSize;

            // Sonuçları Hesapla

            const results = {

                bus: {

                    dbl: priceDbl + perPersonBusCost,

                    trpl: priceTrpl + perPersonBusCost,

                    quad: priceQuad + perPersonBusCost,

                },

                high: {

                    dbl: priceDbl + perPersonHighCost,

                    trpl: priceTrpl + perPersonHighCost,

                    quad: priceQuad + perPersonHighCost,

                },

                gmc: {

                    dbl: priceDbl + perPersonGmcCost,

                    trpl: priceTrpl + perPersonGmcCost,

                    quad: priceQuad + perPersonGmcCost,

                },

                train: {

                    dbl: priceDbl + perPersonTrainCost,

                    trpl: priceTrpl + perPersonTrainCost,

                    quad: priceQuad + perPersonTrainCost,

                }

            };

            // Sonuçları tabloya yazdır

            const tableBody = document.querySelector("#resultsTable tbody");

            tableBody.innerHTML = ''; // Eski sonuçları temizle

            const formatter = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' });

            const createRow = (transport, data) => {

                return `

                    <tr>

                        <td><strong>${transport}</strong></td>

                        <td>${formatter.format(data.dbl)}</td>

                        <td>${formatter.format(data.trpl)}</td>

                        <td>${formatter.format(data.quad)}</td>

                    </tr>

                `;

            };

            tableBody.innerHTML += createRow('Otobüs ile', results.bus);

            tableBody.innerHTML += createRow('"High" Araç ile', results.high);

            tableBody.innerHTML += createRow('GMC Araç ile', results.gmc);

            tableBody.innerHTML += createRow('Hızlı Tren ile', results.train);

           

            // Konaklama süresini göster

            const totalNights = medineNights + mekkeNights;

            document.getElementById('totalNightsDisplay').innerText = `Toplam ${totalNights} Gecelik Program (${medineNights} Gece Medine, ${mekkeNights} Gece Mekke)`;

            // Sonuçlar bölümünü görünür yap

            document.getElementById('results').style.display = 'block';

        }
